# 前端改造说明

## 改造概述

将所有 `wx.cloud.callFunction` 调用改为 HTTP 请求调用。

---

## 1. 创建HTTP请求工具

### 文件位置
`miniprogram/utils/request.js`

### 完整代码
```javascript
/**
 * HTTP请求封装
 * 替代 wx.cloud.callFunction
 */

// API基础地址 - 部署后修改为实际地址
const BASE_URL = 'https://your-domain.com/api';

// 开发环境可以使用本地地址
// const BASE_URL = 'http://localhost:8080/api';

/**
 * 统一请求方法
 * @param {Object} options 请求选项
 * @param {string} options.url 请求路径
 * @param {string} options.method 请求方法
 * @param {Object} options.data 请求数据
 * @param {boolean} options.showLoading 是否显示加载提示
 */
const request = (options = {}) => {
  return new Promise((resolve, reject) => {
    // 获取token
    const token = wx.getStorageSync('token');

    // 显示加载提示
    if (options.showLoading !== false) {
      wx.showLoading({ title: '加载中...' });
    }

    wx.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Authorization': token ? `Bearer ${token}` : '',
        'Content-Type': 'application/json'
      },
      success: (res) => {
        wx.hideLoading();

        if (res.statusCode === 200) {
          const data = res.data;
          if (data.code === 200) {
            resolve(data);
          } else {
            wx.showToast({ title: data.message || '请求失败', icon: 'none' });
            reject(data);
          }
        } else if (res.statusCode === 401) {
          // token过期，清除登录状态
          wx.removeStorageSync('token');
          wx.showToast({ title: '请重新登录', icon: 'none' });
          setTimeout(() => {
            wx.switchTab({ url: '/pages/index/index' });
          }, 1500);
          reject({ code: 401, message: '未登录' });
        } else {
          wx.showToast({ title: '网络错误', icon: 'none' });
          reject({ code: res.statusCode, message: '网络错误' });
        }
      },
      fail: (err) => {
        wx.hideLoading();
        wx.showToast({ title: '网络连接失败', icon: 'none' });
        reject({ code: -1, message: err.errMsg || '网络连接失败' });
      }
    });
  });
};

/**
 * GET请求
 */
const get = (url, data, options = {}) => {
  return request({ url, method: 'GET', data, ...options });
};

/**
 * POST请求
 */
const post = (url, data, options = {}) => {
  return request({ url, method: 'POST', data, ...options });
};

/**
 * PUT请求
 */
const put = (url, data, options = {}) => {
  return request({ url, method: 'PUT', data, ...options });
};

/**
 * DELETE请求
 */
const del = (url, data, options = {}) => {
  return request({ url, method: 'DELETE', data, ...options });
};

module.exports = {
  request,
  get,
  post,
  put,
  del,
  BASE_URL
};
```

## 2. 改造示例

### 2.1 用户登录

**之前**：
```javascript
wx.cloud.callFunction({
  name: 'login',
  success: res => {
    const userInfo = res.result.data;
    app.globalData.userInfo = userInfo;
    app.globalData.isLoggedIn = true;
  }
});
```

**之后**：
```javascript
const { post } = require('../../utils/request');

// 先获取微信code
wx.login({
  success: (loginRes) => {
    post('/user/login', { code: loginRes.code })
      .then(res => {
        const { token, userInfo } = res.data;
        wx.setStorageSync('token', token);
        app.globalData.userInfo = userInfo;
        app.globalData.isLoggedIn = true;
      });
  }
});
```

### 2.2 获取活动列表

**之前**：
```javascript
wx.cloud.callFunction({
  name: 'get-activities',
  data: { type: 'available' },
  success: res => {
    if (res.result.success) {
      this.setData({ activities: res.result.data });
    }
  }
});
```

**之后**：
```javascript
const { get } = require('../../utils/request');

get('/activity/list', { type: 'available' })
  .then(res => {
    this.setData({ activities: res.data });
  });
```

### 2.3 报名活动

**之前**：
```javascript
wx.cloud.callFunction({
  name: 'join-activity',
  data: {
    activityId: this.data.activityId,
    nickname: nickname,
    level: level,
    avatar: avatar
  },
  success: res => {
    if (res.result.success) {
      wx.showToast({ title: '报名成功' });
    } else {
      wx.showToast({ title: res.result.error, icon: 'none' });
    }
  }
});
```

**之后**：
```javascript
const { post } = require('../../utils/request');

post('/registration/join', {
  activityId: this.data.activityId,
  nickname: nickname,
  level: level,
  avatar: avatar
}).then(res => {
  wx.showToast({ title: '报名成功' });
}).catch(err => {
  wx.showToast({ title: err.message, icon: 'none' });
});
```

### 2.4 AI聊天

**之前**：
```javascript
wx.cloud.callFunction({
  name: 'ai-chat',
  data: { message: content },
  success: res => {
    if (res.result && res.result.success) {
      const aiMsg = this._parseMessage({
        role: 'assistant',
        content: res.result.data.content
      });
      this.setData({ messages: [...this.data.messages, aiMsg] });
    }
  }
});
```

**之后**：
```javascript
const { post } = require('../../utils/request');

post('/ai/chat', { message: content })
  .then(res => {
    const aiMsg = this._parseMessage({
      role: 'assistant',
      content: res.data.content
    });
    this.setData({ messages: [...this.data.messages, aiMsg] });
  });
```

## 6. 文件上传改造

### 6.1 之前使用云存储
```javascript
wx.cloud.uploadFile({
  cloudPath: `avatars/${openid}/${Date.now()}.jpg`,
  filePath: tempFilePath,
  success: res => {
    this.setData({ avatar: res.fileID });
  }
});
```

### 6.2 之后使用OSS
```javascript
// 1. 先获取上传凭证
post('/file/upload-token', { type: 'avatar' })
  .then(res => {
    const { uploadUrl, fileKey } = res.data;

    // 2. 上传到OSS
    wx.uploadFile({
      url: uploadUrl,
      filePath: tempFilePath,
      name: 'file',
      formData: { key: fileKey },
      success: uploadRes => {
        const fileUrl = uploadRes.data;
        this.setData({ avatar: fileUrl });
      }
    });
  });
```

```javascript
  // 处理实时更新
});
```
